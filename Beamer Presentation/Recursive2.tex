\begin{frame}{Recursive Formulation}
    \large The final recursion solution:
       
        \small{
            \[  LCS(m,n) = 
                \begin{cases}
                  0 & m == 0 || n == 0\\
                  LCS(m-1 , n-1)+1 & X[m] == Y[n] \\
                  max\{LCS(m-1 ,n),LCS(m,n-1)\}  & X[m] != Y[n]
               \end{cases}
            \]
        }
\end{frame}
